...     use_ldc=True,
...     is_reversible=False,
...     pt_ratio=60.0,
...     ldc_R=9,
...     ldc_X=3,
...     controlled_bus = bus_1,
...     controlled_phase = Phase.A,
...     ct_primary=700,
...     max_step=16,
...     bandwidth=3,  # volts
... )

>>> regulator_1 = DistributionRegulator(
...     name="Regulator-1",
...     buses=[bus_1, bus_2],
...     winding_phases=[[Phase.A], [Phase.A]],
...     equipment=regulator_equipment,
...     controllers=[regulator_controller],
... )
